/**
     * Inserta un producto
     */
    public function update($id) {
        $nombre = $this->input->post('prod_nombre');
        $descripcion = $this->input->post('prod_descripcion');
        
        $sql = "UPDATE producto_bd SET 
                prod_nombre = '$nombre', 
                prod_descripcion = '$descripcion', 
                prod_imagen = null;";
        $return = pg_query($this->conn, $sql);
        
        //Return false if have error
        return !$return;
    }


    /**

    public function edit($id)
    {

        {
            $id = $this->input->post('prod_id');
            $has_error = $this->producto_model->update($id);
            
            if($has_error)
                echo 'Hubo un error: ActualizaciÃ³n fallida';
        }
        else
        {
            $producto = $this->producto_model->get_where_id($id);

        }
    }
